```csharp
// Enable real-time visualization
var visualGraph = new Graph
{
    StartNode = startNode,
    Nodes = new[] { startNode, processNode, endNode },
    Options = new GraphOptions
    {
        EnableVisualization = true,
        EnableRealTimeUpdates = true
    }
};
```
